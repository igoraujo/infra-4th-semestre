VLANs
======

BH_SW_BeloHorizonte
-------------------

enable
conf t
hostname SW_BH

vlan 5
name TI
exit

vlan 10
name RECEPCAO
exit

vlan 20
name COMERCIAL
exit

vlan 30
name FINANCEIRO
exit

vlan 40
name CONTABILIDADE
exit

int f 0/24
switchport mode trunk
exit

int f 0/1
switchport mode trunk
exit

int f 0/5
switchport mode trunk
exit

int f 0/10
switchport mode trunk
exit

int f 0/15
switchport mode trunk
exit

vtp mode server
vtp domain SW_BH
vtp password senha123
vtp version 2
exit

exit
